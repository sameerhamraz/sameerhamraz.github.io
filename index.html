import customtkinter as ctk from threading import Thread from jarvis_core import jarvis_brain # هسته JARVIS ctk.set_appearance_mode("dark") ctk.set_default_color_theme("dark-blue") app = ctk.CTk() app.geometry("500x350") app.title("J.A.R.V.I.S") title = ctk.CTkLabel( app, text="J.A.R.V.I.S", font=("Orbitron", 32), text_color="#00FFFF" ) title.pack(pady=30) status = ctk.CTkLabel( app, text="STATUS: ONLINE", font=("Arial", 16), text_color="lime" ) status.pack(pady=10) def start_jarvis(): Thread(target=jarvis_brain).start() start_btn = ctk.CTkButton( app, text="▶ ACTIVATE", command=start_jarvis, fg_color="#00FFFF", text_color="black", font=("Arial", 16) ) start_btn.pack(pady=40) app.mainloop()
